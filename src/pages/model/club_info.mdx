# Club-Info

Das Club-Modell bietet dir detailliertere Informationen, mit welchen du Informationen über jedes Team abrufen kannst.

## Json-Beispiel

 ```json

"fussballDeUrl": "/mannschaft/borussia-dortmund-borussia-dortmund-westfalen/-/saison/2122/team-id/011MI8V6UC000000VTVG0001VTR8C1K7",
"name": "Herren - Borussia Dortmund",
"urls": {
    "nextGames": "/api/team/next_games/011MI8V6UC000000VTVG0001VTR8C1K7",
    "prevGames": "/api/team/prev_games/011MI8V6UC000000VTVG0001VTR8C1K7",
    "table": "/api/team/table/011MI8V6UC000000VTVG0001VTR8C1K7",
    "allInfo": "/api/team/011MI8V6UC000000VTVG0001VTR8C1K7"
}
```

---

## Eigenschaften

<Properties>
  <Property name="fussballDeUrl" type="string">
      URL auf Fussball.de
  </Property>
  <Property name="name" type="string">
    Team Name
  </Property>
  <Property name="urls" type="json">
      Das Club-Modell enthält die URLs, über die du auf detaillierte Informationen über das Team zugreifen kannst.
  </Property>
  <Property name="nextGames" type="string">
      URL zum nächsten Spiel des Teams
  </Property>
  <Property name="prevGames" type="string">
      URL zum letzten Spiel des Teams
  </Property>
  <Property name="table" type="string">
      URL zur aktuellen Ligatabelle des Teams
  </Property>
  <Property name="allInfo" type="string">
      URL, wo du alle Informationen auf einmal sehen kannst. Es enthält Informationen von den letzten Spielen, den nächsten Spielen und der aktuellen Tabelle.
  </Property>
</Properties>
